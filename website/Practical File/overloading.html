1) Create a class Fraction to represent fractions. Overload the + operator to add two fractions.

#include <iostream.h>

class Fraction {
public:
   int num, den;

   Fraction(int n=0, int d=1) {
       num = n;
       den = d;
   }

   Fraction operator+(Fraction f) {
       Fraction temp;
       temp.num = num*f.den + f.num*den;
       temp.den = den*f.den;
       return temp;
   }

   void display() {
       cout << num << "/" << den;
   }
};

int main() {
   Fraction f1(1,2), f2(3,4), f3;
   f3 = f1 + f2;
   f3.display();
   return 0;
}

2) Implement a class String to represent strings. Overload the + operator to concatenate two strings.

#include <iostream.h>
#include <string.h>

class String {
public:
   char str[100];

   String() {
       str[0] = '\0';
   }

   String(char s[]) {
       strcpy(str, s);
   }

   String operator+(String s) {
       String temp;
       strcpy(temp.str, str);
       strcat(temp.str, s.str);
       return temp;
   }

   void display() {
       cout << str;
   }
};

int main() {
   String s1("Hello "), s2("World"), s3;
   s3 = s1 + s2;
   s3.display();
   return 0;
}

4) Define a class Matrix to represent matrices. Overload the * operator to multiply two matrices.

#include <iostream.h>

class Matrix {
public:
   int a[2][2];

   Matrix operator*(Matrix m) {
       Matrix temp;
       for(int i=0;i<2;i++) {
           for(int j=0;j<2;j++) {
               temp.a[i][j]=0;
               for(int k=0;k<2;k++) {
                   temp.a[i][j]+=a[i][k]*m.a[k][j];
               }
           }
       }
       return temp;
   }

   void display() {
       for(int i=0;i<2;i++) {
           for(int j=0;j<2;j++) {
               cout << a[i][j] << " ";
           }
           cout << endl;
       }
   }
};

int main() {
   Matrix m1, m2, m3;

   m1.a[0][0]=1; m1.a[0][1]=2;
   m1.a[1][0]=3; m1.a[1][1]=4;

   m2.a[0][0]=5; m2.a[0][1]=6;
   m2.a[1][0]=7; m2.a[1][1]=8;

   m3 = m1 * m2;
   m3.display();
   return 0;
}

5) Define a class Date to represent dates. Overload the < operator to compare two dates.

#include <iostream.h>

class Date {
public:
   int d, m, y;

   Date(int d1=0,int m1=0,int y1=0) {
       d=d1; m=m1; y=y1;
   }

   int operator<(Date dt) {
       if(y < dt.y) return 1;
       if(y == dt.y && m < dt.m) return 1;
       if(y == dt.y && m == dt.m && d < dt.d) return 1;
       return 0;
   }
};

int main() {
   Date d1(1,1,2020), d2(2,2,2021);
   if(d1 < d2)
       cout << "d1 is earlier";
   else
       cout << "d2 is earlier";
   return 0;
}

7) Create a class Rectangle to represent rectangles. Overload the == operator to check if two rectangles are equal.

#include <iostream.h>

class Rectangle {
public:
   int l, b;

   Rectangle(int x=0,int y=0) {
       l=x; b=y;
   }

   int operator==(Rectangle r) {
       return (l==r.l && b==r.b);
   }
};

int main() {
   Rectangle r1(5,4), r2(5,4);
   if(r1 == r2)
       cout << "Equal";
   else
       cout << "Not Equal";
   return 0;
}

8) Define a class Time to represent time. Overload the << operator to display the time in HH:MM format.

#include <iostream.h>

class Time {
public:
   int h, m;

   Time(int h1=0,int m1=0) {
       h=h1; m=m1;
   }

   friend ostream& operator<<(ostream &out, Time t) {
       out << t.h << ":" << t.m;
       return out;
   }
};

int main() {
   Time t(10,30);
   cout << t;
   return 0;
}
