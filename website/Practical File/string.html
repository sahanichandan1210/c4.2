19) Program to find all the permutations of a string

#include <iostream.h>
#include <conio.h>
#include <string.h>

void perm(char *str, int l, int r) {
   int i;
   char temp;
   if(l==r)
       cout << str << " ";
   else {
       for(i=l;i<=r;i++) {
           temp=str[l]; str[l]=str[i]; str[i]=temp;
           perm(str,l+1,r);
           temp=str[l]; str[l]=str[i]; str[i]=temp;
       }
   }
}

void main() {
   char str[20];
   cin >> str;
   perm(str,0,strlen(str)-1);
   getch();
}

31) Program to print smallest and biggest possible palindrome word in a given string

#include <iostream.h>
#include <conio.h>
#include <string.h>

int isPal(char w[]) {
   int i=0, j=strlen(w)-1;
   while(i<j) {
       if(w[i]!=w[j]) return 0;
       i++; j--;
   }
   return 1;
}

void main() {
   char str[100], word[20], minw[20], maxw[20];
   int i=0, j=0, len, min=100, max=0;

   cin.getline(str,100);

   while(1) {
       if(str[i]==' ' || str[i]=='\0') {
           word[j]='\0';
           if(isPal(word)) {
               len = strlen(word);
               if(len<min) { min=len; strcpy(minw,word); }
               if(len>max) { max=len; strcpy(maxw,word); }
           }
           j=0;
           if(str[i]=='\0') break;
       } else word[j++]=str[i];
       i++;
   }

   cout << minw << " " << maxw;
   getch();
}
