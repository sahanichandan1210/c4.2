11) C++ Program to convert Decimal to Binary

#include <iostream.h>

int main() {
   int num = 10, binary[32], i = 0;

   while(num > 0) {
       binary[i++] = num % 2;
       num /= 2;
   }

   for(int j = i - 1; j >= 0; j--) {
       cout << binary[j];
   }
   return 0;
}

12) C++ Program to convert Decimal to Hex

#include <iostream.h>

int main() {
   int num = 255;
   char hex[20];
   int i = 0;

   while(num != 0) {
       int rem = num % 16;
       if(rem < 10)
           hex[i++] = rem + 48;
       else
           hex[i++] = rem + 55;
       num /= 16;
   }

   for(int j = i - 1; j >= 0; j--) {
       cout << hex[j];
   }
   return 0;
}

13) C++ Program to convert Binary to Decimal

#include <iostream.h>
#include <math.h>

int main() {
   int binary = 1010, decimal = 0, i = 0, rem;

   while(binary != 0) {
       rem = binary % 10;
       decimal += rem * pow(2, i);
       binary /= 10;
       i++;
   }

   cout << decimal;
   return 0;
}

14) C++ Program to convert Hex to Decimal

#include <iostream.h>
#include <math.h>
#include <string.h>

int main() {
   char hex[] = "1A";
   int len = strlen(hex);
   int base = 1, decimal = 0;

   for(int i = len - 1; i >= 0; i--) {
       if(hex[i] >= '0' && hex[i] <= '9')
           decimal += (hex[i] - 48) * base;
       else if(hex[i] >= 'A' && hex[i] <= 'F')
           decimal += (hex[i] - 55) * base;
       base *= 16;
   }

   cout << decimal;
   return 0;
}

15) C++ Program to convert Octal to Decimal

#include <iostream.h>
#include <math.h>

int main() {
   int octal = 17, decimal = 0, i = 0, rem;

   while(octal != 0) {
       rem = octal % 10;
       decimal += rem * pow(8, i);
       octal /= 10;
       i++;
   }

   cout << decimal;
   return 0;
}
