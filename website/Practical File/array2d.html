2) Program to Multiply Two Matrices

#include <iostream.h>
#include <conio.h>
void main() {
   int a[10][10], b[10][10], c[10][10], r1, c1, r2, c2, i, j, k;

   cin >> r1 >> c1;
   cin >> r2 >> c2;

   for(i=0;i<r1;i++)
       for(j=0;j<c1;j++)
           cin >> a[i][j];

   for(i=0;i<r2;i++)
       for(j=0;j<c2;j++)
           cin >> b[i][j];

   for(i=0;i<r1;i++) {
       for(j=0;j<c2;j++) {
           c[i][j] = 0;
           for(k=0;k<c1;k++)
               c[i][j] += a[i][k] * b[k][j];
       }
   }

   for(i=0;i<r1;i++) {
       for(j=0;j<c2;j++)
           cout << c[i][j] << " ";
       cout << endl;
   }

   getch();
}

9) Program to find the frequency of odd & even numbers in the given matrix

#include <iostream.h>
#include <conio.h>
void main() {
   int a[10][10], r, c, i, j, even=0, odd=0;

   cin >> r >> c;

   for(i=0;i<r;i++)
       for(j=0;j<c;j++)
           cin >> a[i][j];

   for(i=0;i<r;i++)
       for(j=0;j<c;j++) {
           if(a[i][j] % 2 == 0)
               even++;
           else
               odd++;
       }

   cout << even << " " << odd;

   getch();
}

10) Program to determine whether a given matrix is an identity matrix

   for(i=0;i<n;i++)
       for(j=0;j<n;j++)
           cin >> a[i][j];

   for(i=0;i<n;i++) {
       for(j=0;j<n;j++) {
           if(i==j && a[i][j]!=1)
               flag = 0;

               flag = 0;
       }
   }

   cout << flag;

   getch();
}

12) Program to determine whether a given matrix is a sparse matrix

#include <iostream.h>
#include <conio.h>
void main() {
   int a[10][10], r, c, i, j, zero=0;

   cin >> r >> c;

   for(i=0;i<r;i++)
       for(j=0;j<c;j++)
           cin >> a[i][j];

   for(i=0;i<r;i++)
       for(j=0;j<c;j++) {
           if(a[i][j] == 0)
               zero++;
       }

   if(zero > (r*c)/2)
       cout << 1;
   else
       cout << 0;

   getch();
}
