// 6) Bank and Account
class Account {
   int accNo;
   float balance;
public:
   void create(int a, float b) {
       accNo = a;
       balance = b;
   }
   void deposit(float amt) { balance += amt; }
   void withdraw(float amt) { balance -= amt; }
   void display() {
       cout << "Account: " << accNo << ", Balance: " << balance << endl;
   }
};

class Bank {
   Account acc[5];
   int count;
public:
   Bank() { count = 0; }
   void addAccount(int a, float b) {
       acc[count].create(a, b);
       count++;
   }
   void removeAccount() {
       if(count > 0) count--;
   }
   void deposit(int i, float amt) {
       acc[i].deposit(amt);
   }
   void withdraw(int i, float amt) {
       acc[i].withdraw(amt);
   }
   void display() {
       for(int i=0;i<count;i++) acc[i].display();
   }
};

// 18) Shape (Abstract)
class Shape {
public:
   virtual float area() = 0;
   virtual float perimeter() = 0;
};

class Rectangle : public Shape {
   float l, w;
public:
   Rectangle(float a, float b) { l = a; w = b; }
   float area() { return l * w; }
   float perimeter() { return 2 * (l + w); }
};

class Circle : public Shape {
   float r;
public:
   Circle(float x) { r = x; }
   float area() { return 3.14 * r * r; }
   float perimeter() { return 2 * 3.14 * r; }
};

class Triangle : public Shape {
   float a, b, c;
public:
   Triangle(float x, float y, float z) {
       a = x; b = y; c = z;
   }
   float area() { return (a + b + c) / 2; }
   float perimeter() { return a + b + c; }
};

// 19) Restaurant
class Restaurant {
   char items[10][20];
   float price[10], rating[10];
   int count;
public:
   Restaurant() { count = 0; }
   void addItem(char i[], float p, float r) {
       strcpy(items[count], i);
       price[count] = p;
       rating[count] = r;
       count++;
   }
   void removeItem() {
       if(count > 0) count--;
   }
   float avgRating() {
       float sum = 0;
       for(int i=0;i<count;i++) sum += rating[i];
       return sum / count;
   }
};

// 31) Student with validation
class Student {
   int student_id;
   char student_name[20];
   int grades[10];
   int count;
public:
   Student() { count = 0; }

   void setId(int id) { student_id = id; }
   void setName(char n[]) { strcpy(student_name, n); }

   int getId() { return student_id; }
   char* getName() { return student_name; }

   void addGrade(int g) {
       if(g >= 0 && g <= 100 && count < 10) {
           grades[count++] = g;
       }
   }

   void displayGrades() {
       for(int i=0;i<count;i++)
           cout << grades[i] << " ";
       cout << endl;
   }
};

int main() {
   Student s;
   s.setId(1);
   s.setName("Aman");

   s.addGrade(85);
   s.addGrade(90);
   s.addGrade(105);

   cout << s.getId() << " " << s.getName() << endl;
   s.displayGrades();

   return 0;
}
