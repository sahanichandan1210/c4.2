Program 7: Decimal to Binary

Problem Statement: Write a C++ program to convert a decimal number to binary.

#include
< iostream.h >
int main() {
   int n, binary[32], i = 0;

   cout << "Enter decimal number: ";
   cin >> n;

   while(n > 0) {
       binary[i] = n % 2;
       n = n / 2;
       i++;

Program 8: Binary to Decimal

Problem Statement: Write a C++ program to convert a binary number to decimal.

#include
< iostream.h >
int main() {
   int binary, decimal = 0, i = 0, rem;

   cout << "Enter binary number: ";
   cin >> binary;

   while(binary != 0) {
       rem = binary % 10;
       decimal += rem * pow(2, i);
       binary /= 10;
       i++;
   }

   cout << "Decimal = " << decimal;
   return 0;
}

Output:

Enter binary number: 101

Decimal = 5

Explanation: Each digit is multiplied by powers of 2 and added to get decimal value.

Program 20: Sum of First 100 Prime Numbers

Problem Statement: Write a C++ program to compute sum of first 100 prime numbers.

#include

int main() {
   int count = 0, num = 2, sum = 0;

   while(count < 100) {
       int flag = 1;

       for(int i = 2; i <= num / 2; i++) {
           if(num % i == 0) {
               flag = 0;
               break;
           }
       }

       if(flag == 1) {
           sum += num;
           count++;
       }

       num++;
   }

   cout << "Sum = " << sum;
   return 0;
}

Output:

Sum = 24133

Explanation: Loop finds prime numbers and adds first 100 primes.

Program 21: Palindrome Number

Problem Statement: Check whether a number is palindrome or not.

#include

int main() {
   int num, original, reverse = 0, rem;

   cout << "Enter number: ";
   cin >> num;

   original = num;

   while(num != 0) {
       rem = num % 10;
       reverse = reverse * 10 + rem;
       num /= 10;
   }

   if(original == reverse)
       cout << "Palindrome";
   else
       cout << "Not Palindrome";

   return 0;
}

Explanation: Reverse of number is compared with original.

104) Palindrome Number (Method)

#include <iostream.h>
#include <conio.h>

int palindrome(int n) {
   int rev=0, rem, temp=n;

   while(n!=0) {
       rem=n%10;
       rev=rev*10+rem;
       n=n/10;
   }

   if(temp==rev) return 1;
   else return 0;
}

int main() {
   int n;
   cin >> n;

   if(palindrome(n))
       cout << "Palindrome";
   else
       cout << "Not Palindrome";

   getch();
}

111) Roman to Integer

#include <iostream.h>
#include <conio.h>
#include <string.h>

int value(char r) {
   if(r=='I') return 1;
   if(r=='V') return 5;
   if(r=='X') return 10;
   if(r=='L') return 50;
   if(r=='C') return 100;
   if(r=='D') return 500;
   if(r=='M') return 1000;
   return 0;
}

int main() {
   char str[50];
   int i, res=0;

   cin >> str;

   for(i=0; str[i]!='\0'; i++) {
       if(value(str[i]) < value(str[i+1]))
           res -= value(str[i]);
       else
           res += value(str[i]);
   }

   cout << res;

   getch();
}
