2) Write a C++ program to count the number of lines in a text file.

#include <iostream.h>
#include <fstream.h>

int main() {
   ifstream file("test.txt");
   char ch;
   int lines = 0;

   while(file.get(ch)) {
       if(ch == '\n')
           lines++;
   }

   file.close();
   cout << lines;
   return 0;
}

7) Write a C++ program to sort the lines of a text file in alphabetical order.

#include <iostream.h>
#include <fstream.h>
#include <string.h>

int main() {
   char lines[10][100], temp[100];
   int count = 0;

   ifstream file("test.txt");

   while(file.getline(lines[count],100)) {
       count++;
   }

   for(int i=0;i<count;i++) {
       for(int j=i+1;j<count;j++) {
           if(strcmp(lines[i], lines[j]) > 0) {
               strcpy(temp, lines[i]);
               strcpy(lines[i], lines[j]);
               strcpy(lines[j], temp);
           }
       }
   }

   for(int i=0;i<count;i++)
       cout << lines[i] << endl;

   return 0;
}

10) Write a C++ program to split a large text file into smaller files of equal size

#include <iostream.h>
#include <fstream.h>

int main() {
   ifstream file("test.txt");
   ofstream f1("part1.txt"), f2("part2.txt");
   char ch;
   int count = 0;

   while(file.get(ch)) {
       if(count % 2 == 0)
           f1.put(ch);
       else
           f2.put(ch);
       count++;
   }

   file.close();
   f1.close();
   f2.close();
   return 0;
}

16) Write a C++ program to store the information of 10 Students having rollno,name,marks,mobileno properties with operations

#include <iostream.h>
#include <fstream.h>

class Student {
public:
   int roll;
   char name[50];
   float marks;
   char mobile[15];

   void get() {
       cin >> roll >> name >> marks >> mobile;
   }

   void show() {
       cout << roll << " " << name << " " << marks << " " << mobile << endl;
   }
};

int main() {
   Student s;
   fstream file("stu.dat", ios::binary | ios::out);

   for(int i=0;i<10;i++) {
       s.get();
       file.write((char*)&s, sizeof(s));
   }

   file.close();

   fstream file2("stu.dat", ios::binary | ios::in);
   while(file2.read((char*)&s, sizeof(s))) {
       s.show();
   }

   file2.close();
   return 0;
}
